print_lock = threading.Lock()

def log(name, msg):

def banner(msg):

def run(cmd, env=None, cwd=None, verbose=False):
    merged = os.environ.copy()
    if env: merged.update(env)
    if verbose:
        return subprocess.run(cmd, shell=True, env=merged, cwd=cwd, text=True)
    return subprocess.run(cmd, shell=True, env=merged, cwd=cwd, text=True, capture_output=True)

def run_output(cmd, env=None, cwd=None, timeout=None):
    merged = os.environ.copy()
    if env: merged.update(env)
    try:
        r = subprocess.run(cmd, shell=True, env=merged, cwd=cwd,
                           capture_output=True, text=True, timeout=timeout)
        return r.stdout.strip()
    except subprocess.TimeoutExpired:
        return ""

def parse_json_field(raw, field):
    try: return json.loads(raw).get(field, "")
    except Exception:
        for line in raw.splitlines():
            try:
                d = json.loads(line)
                if field in d: return d[field]
            except Exception: pass
    return ""

def parse_json_safe(raw):
    try: return json.loads(raw)
    except Exception:
        for line in raw.splitlines():
            try:
                d = json.loads(line)
                if isinstance(d, dict): return d
            except Exception: pass
    return {}

def get_eth_balance(address):
    """Return balance in ETH (float)."""
    w3 = get_web3()
    if not w3: return 0
    try:
        wei = w3.eth.get_balance(Web3.to_checksum_address(address))
        return w3.from_wei(wei, 'ether')
    except Exception as e:
        return 0


def check_already_registered(address):
    """
    Cek apakah wallet sudah punya recipient di AWP Registry.
    
    Panggil contract call: getRecipient(address) → address
    Kalau return non-zero → sudah registered.
    
    ABI minimal: selector getRecipient(address) = keccak256("getRecipient(address)")[:4]
    """
    from web3 import Web3
    w3 = get_web3()
    if not w3: return False
    
    try:
        # Minimal ABI — cukup getRecipient
        abi = [{
            "inputs": [{"name": "user", "type": "address"}],
            "name": "getRecipient",
            "outputs": [{"name": "", "type": "address"}],
            "stateMutability": "view",
            "type": "function"
        }]
        contract = w3.eth.contract(
            address=Web3.to_checksum_address(AWP_REGISTRY_CONTRACT),
            abi=abi
        )
        recipient = contract.functions.getRecipient(
            Web3.to_checksum_address(address)
        ).call()
        return recipient != "0x0000000000000000000000000000000000000000"
    except Exception as e:
        # getRecipient mungkin beda nama. Coba cara lain: read event logs
        # atau cek via eth_call raw dengan selector yang benar
        pass
    
    # Fallback: cek via event logs RecipientSet untuk address ini
    try:
        filter_params = {
            'address': Web3.to_checksum_address(AWP_REGISTRY_CONTRACT),
            'topics': [
                EVENT_RECIPIENT_SET,
                '0x' + '0' * 24 + address.lower().replace('0x', ''),  # addr as topic1 (padded)
            ],
            'fromBlock': 0,
            'toBlock': 'latest'
        }
        logs = w3.eth.get_logs(filter_params)
        return len(logs) > 0
    except Exception:
        return False


def fund_wallet(recipient_addr, amount_eth, wallet_name):
    """Send ETH dari FUNDER ke recipient_addr."""
    from web3 import Web3
    from eth_account import Account
    
    w3 = get_web3()
    if not w3:
        log(wallet_name, "❌ Web3 not connected")
        return None
    
    if not FUNDER_PRIVATE_KEY:
        log(wallet_name, "❌ FUNDER_PRIVATE_KEY not set, cannot fund!")
        return None
    
    try:
        funder = Account.from_key(FUNDER_PRIVATE_KEY)
        funder_addr = funder.address
        
        # Check balance
        balance_wei = w3.eth.get_balance(funder_addr)
        needed_wei = w3.to_wei(amount_eth, 'ether')
        if balance_wei < needed_wei:
            log(wallet_name,
                f"❌ Funder balance insufficient: "
                f"{w3.from_wei(balance_wei, 'ether')} ETH < {amount_eth} ETH")
            return None
        
        # Check recipient current balance — kalau sudah cukup, skip
        recip_balance = w3.eth.get_balance(Web3.to_checksum_address(recipient_addr))
        if recip_balance >= needed_wei // 2:  # minimal 50% of target
            log(wallet_name, f"Already funded ({w3.from_wei(recip_balance,'ether'):.6f} ETH), skip")
            return "skipped"
        
        # Build tx
        nonce = w3.eth.get_transaction_count(funder_addr, 'pending')
        tx = {
            'to': Web3.to_checksum_address(recipient_addr),
            'value': needed_wei,
            'gas': 21000,
            'maxFeePerGas': w3.to_wei(0.01, 'gwei'),
            'maxPriorityFeePerGas': w3.to_wei(0.001, 'gwei'),
            'nonce': nonce,
            'chainId': AWP_CHAIN_ID,
            'type': 2,
        }
        
        signed = funder.sign_transaction(tx)
        tx_hash = w3.eth.send_raw_transaction(signed.raw_transaction)
        tx_hex = tx_hash.hex()
        log(wallet_name, f"Fund tx: 0x{tx_hex[:20]}... waiting...")
        
        receipt = w3.eth.wait_for_transaction_receipt(tx_hash, timeout=120)
        if receipt.status == 1:
            log(wallet_name, f"✅ Funded {amount_eth} ETH")
            return tx_hex
        else:
            log(wallet_name, f"❌ Fund tx failed: {tx_hex}")
            return None
    except Exception as e:
        log(wallet_name, f"Fund error: {e}")
        return None


def call_set_recipient(wallet_private_key, recipient_addr, wallet_name):
    """
    Call setRecipient(recipient_addr) on AWP Registry.
    
    Pakai wallet_private_key untuk sign dan send tx.
    recipient_addr biasanya = self address (solo mining Option A).
    """
    from web3 import Web3
    from eth_account import Account
    
    w3 = get_web3()
    if not w3:
        return None
    
    try:
        account = Account.from_key(wallet_private_key)
        addr = account.address
        
        # Check balance
        balance = w3.eth.get_balance(addr)
        if balance < w3.to_wei(0.00003, 'ether'):  # minimum buat gas
            log(wallet_name, f"❌ Insufficient gas: {w3.from_wei(balance,'ether'):.6f} ETH")
            return None
        
        # Build calldata: selector (4 bytes) + padded recipient address (32 bytes)
        recipient_clean = recipient_addr.lower().replace("0x", "")
        if len(recipient_clean) != 40:
            log(wallet_name, f"❌ Invalid recipient address length")
            return None
        
        calldata = AWP_SET_RECIPIENT_SELECTOR + ("0" * 24) + recipient_clean
        
        # Build tx
        nonce = w3.eth.get_transaction_count(addr, 'pending')
        
        # Estimate gas (dengan safety margin)
        try:
            est_gas = w3.eth.estimate_gas({
                'from': addr,
                'to': Web3.to_checksum_address(AWP_REGISTRY_CONTRACT),
                'data': calldata,
                'value': 0,
            })
            gas_limit = int(est_gas * 1.3)  # 30% margin
        except Exception as e:
            log(wallet_name, f"Gas estimate failed: {e}, use default 100k")
            gas_limit = 100000
        
        tx = {
            'to': Web3.to_checksum_address(AWP_REGISTRY_CONTRACT),
            'value': 0,
            'data': calldata,
            'gas': gas_limit,
            'maxFeePerGas': w3.to_wei(0.01, 'gwei'),
            'maxPriorityFeePerGas': w3.to_wei(0.001, 'gwei'),
            'nonce': nonce,
            'chainId': AWP_CHAIN_ID,
            'type': 2,
        }
        
        signed = account.sign_transaction(tx)
        tx_hash = w3.eth.send_raw_transaction(signed.raw_transaction)
        tx_hex = tx_hash.hex()
        log(wallet_name, f"setRecipient tx: 0x{tx_hex[:20]}... waiting receipt...")
        
        receipt = w3.eth.wait_for_transaction_receipt(tx_hash, timeout=180)
        if receipt.status == 1:
            log(wallet_name, f"✅ setRecipient SUCCESS: https://basescan.org/tx/0x{tx_hex}")
            # Verify event emitted
            for log_entry in receipt.logs:
                if (log_entry.address.lower() == AWP_REGISTRY_CONTRACT.lower() and
                    log_entry.topics and
                    log_entry.topics[0].hex().lower() == EVENT_RECIPIENT_SET.lower().replace("0x", "").replace("0x", "")):
                    log(wallet_name, "✓ RecipientSet event emitted")
                    break
            return tx_hex
        else:
            log(wallet_name, f"❌ setRecipient tx reverted: 0x{tx_hex}")
            return None
    except Exception as e:
        log(wallet_name, f"setRecipient error: {e}")
        return None

    else:

def register_wallet_onchain(wallet_name, address, mnemonic, hd_index):
    """
    MAIN REGISTRATION FUNCTION — on-chain langsung.
    
    Flow:
      1. Cek apakah sudah registered (skip kalau ya)
      2. Cek balance ETH; kalau < threshold → fund dari FUNDER
      3. Derive private key dari mnemonic
      4. Call setRecipient(self_address) on-chain
      5. Verify via getRecipient atau event log
    """
    # Step 1: Check sudah registered?
    log(wallet_name, "Checking on-chain registration...")
    if check_already_registered(address):
        log(wallet_name, "✅ Already registered on-chain")
        return True
    
    # Step 2: Ensure wallet has ETH for gas
    balance = get_eth_balance(address)
    min_balance = 0.00005  # 0.00005 ETH minimum untuk 1 tx
    log(wallet_name, f"Balance: {balance:.6f} ETH")
    
    if float(balance) < min_balance:
        if not FUNDER_PRIVATE_KEY:
            log(wallet_name, "❌ Need ETH but FUNDER_PRIVATE_KEY not set!")
            log(wallet_name, f"   Manual: send {ETH_PER_WALLET} ETH Base ke {address}")
            return False
        
        log(wallet_name, f"Funding from FUNDER ({ETH_PER_WALLET} ETH)...")
        tx = fund_wallet(address, ETH_PER_WALLET, wallet_name)
        if not tx:
            return False
        time.sleep(3)
    
    # Step 3: Derive private key
    try:
        privkey = derive_private_key(mnemonic, hd_index)
    except Exception as e:
        log(wallet_name, f"❌ Key derive error: {e}")
        return False
    
    # Step 4: Call setRecipient(self)
    log(wallet_name, f"Calling setRecipient({address[:10]}...) on {AWP_REGISTRY_CONTRACT[:10]}...")
    tx_hash = call_set_recipient(privkey, address, wallet_name)
    if not tx_hash:
        return False
    
    # Step 5: Verify
    time.sleep(5)
    if check_already_registered(address):
        log(wallet_name, "✅ Registration VERIFIED on-chain!")
        return True
    
    log(wallet_name, "⚠ Tx success but verification failed, assume OK")
    return True

def create_wallets_file(p, count):
    print(f"\n  Generating {count} wallet...")
    lines = []
    for i in range(count):
        name = f"wallet{i+1}"
        m = generate_mnemonic()
        lines.append(f"{name}|{m}")
        print(f"  ✓ {name}")
    p.parent.mkdir(parents=True, exist_ok=True)
    with open(p, "w") as f:
        f.write(f"# Auto-generated {datetime.now()}\n# Format: name|mnemonic\n\n")
        for l in lines: f.write(l + "\n")
    print(f"  ⚠ BACKUP {p}")

def load_wallets_from_file():
    import re
    p = Path(WALLETS_FILE)
    if not p.exists():
        create_wallets_file(p, WALLET_COUNT)
    wallets = []
    with open(p) as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#"): continue
            parts = line.split("|", 1)
            if len(parts) != 2: continue
            n, m = parts[0].strip(), parts[1].strip()
            if not re.match(r'^[a-zA-Z0-9_-]{1,64}$', n): continue
            if len(m.split()) not in (12, 24): continue
            wallets.append((n, m))
    if len(wallets) < WALLET_COUNT:
        need = WALLET_COUNT - len(wallets)
        with open(p, "a") as f:
            for j in range(need):
                idx = len(wallets)+j+1
                n = f"wallet{idx}"; m = generate_mnemonic()
                f.write(f"{n}|{m}\n"); wallets.append((n, m))
    return wallets[:WALLET_COUNT]

def copy_venv(mine_dir, wn):

    if dest.exists() and py.exists() and "3.11" in run_output(f'"{py}" --version 2>/dev/null'):
        return
    if dest.exists(): shutil.rmtree(str(dest), ignore_errors=True)
    src = _template_dir/".venv"
    if not src.exists(): return
    r = run(f'cp -a "{src}" "{dest}"')
    if r.returncode != 0:
        shutil.copytree(str(src), str(dest), symlinks=True)

def import_wallet(name, mnemonic, agent_id):
    env = set_env(agent_id, name)
    raw = run_output(f'AWP_AGENT_ID="{agent_id}" awp-wallet receive', env=env)
    addr = parse_json_field(raw, "eoaAddress")
    if addr:
        log(name, f"Already imported → {addr}")
        return addr
    log(name, "Importing...")
    r = run(f'AWP_AGENT_ID="{agent_id}" awp-wallet import --mnemonic "{mnemonic}"', env=env)
    if r.returncode != 0:
        log(name, f"Import err: {(r.stderr or r.stdout)[:120]}")
        return None
    raw = run_output(f'AWP_AGENT_ID="{agent_id}" awp-wallet receive', env=env)
    return parse_json_field(raw, "eoaAddress")

def clone_mine_skill(mine_dir, name):
    if not mine_dir.exists():
        run(f'git clone https://github.com/awp-worknet/mine-skill "{mine_dir}" -q')
    else:
        if run(f'git -C "{mine_dir}" pull --ff-only -q').returncode != 0:
            run(f'git -C "{mine_dir}" fetch origin')
            run(f'git -C "{mine_dir}" reset --hard origin/main')

def python_bin_of(mine_dir):
    py = mine_dir/".venv"/"bin"/"python"
    return str(py) if py.exists() else PYTHON311


def setup_wallet_worker(args):
    name, mnemonic, hd_idx, total = args
    agent_id = name
    mine_dir = Path(HOME)/f"mine-skill-{name}"
    log(name, f"Setup starting...")

    address = import_wallet(name, mnemonic, agent_id)
    if not address: return None

    clone_mine_skill(mine_dir, name)
    copy_venv(mine_dir, name)

    env = set_env(agent_id, name)
    py_bin = python_bin_of(mine_dir)

    # Unlock wallet token
    raw = run_output(f'AWP_AGENT_ID="{agent_id}" awp-wallet unlock --scope full --duration 86400', env=env)
    token = parse_json_field(raw, "sessionToken")
    if not token:
        log(name, f"Token FAIL: {raw[:100]}")
        return None
    log(name, f"Token OK: {token[:20]}...")
    env["AWP_WALLET_TOKEN"] = token

    # Init mine-skill
    log(name, "mine-skill init...")
    run_output(f'"{py_bin}" scripts/run_tool.py init', env=env, cwd=str(mine_dir), timeout=60)

    # 🔑 ON-CHAIN REGISTRATION (the fix!)
    log(name, "═══ ON-CHAIN REGISTRATION ═══")
    # Untuk mode 'file' (WALLET_MODE="file"): mnemonic beda per wallet, pakai default derive (index None)
    # Untuk mode 'single': satu mnemonic, pakai hd_idx
    derive_idx = hd_idx if WALLET_MODE == "single" else None
    registered = register_wallet_onchain(name, address, mnemonic, derive_idx)

    ip = run_output("curl -s --max-time 5 https://api.ipify.org", env=env)
    log(name, f"Setup done | {address} | IP={ip} | Reg={registered}")

    return {
        "name": name, "agent_id": agent_id, "directory": str(mine_dir),
        "address": address, "python_bin": py_bin, "token": token,
        "ip": ip, "mnemonic": mnemonic, "wallet_index": hd_idx,
        "registered": registered,
    }

def unlock_token(w):
    env = set_env(w["agent_id"], w["name"])
    raw = run_output(f'AWP_AGENT_ID="{w["agent_id"]}" awp-wallet unlock --scope full --duration 86400', env=env)
    return parse_json_field(raw, "sessionToken")

def agent_status(py_bin, wd, env):
    return parse_json_safe(run_output(f'"{py_bin}" scripts/run_tool.py agent-control status', env=env, cwd=wd))

def agent_start(py_bin, wd, env, ds=None):
    cmd = f'"{py_bin}" scripts/run_tool.py agent-start'
    if ds: cmd += " " + " ".join(ds)
    return parse_json_safe(run_output(cmd, env=env, cwd=wd))

def agent_stop(py_bin, wd, env):
    return parse_json_safe(run_output(f'"{py_bin}" scripts/run_tool.py agent-control stop', env=env, cwd=wd))


def start_mining(w):
    name, wd, py_bin = w["name"], w["directory"], w["python_bin"]
    last_refresh = time.time()
    
    while True:
        try:
            env = set_env(w["agent_id"], name)
            token = unlock_token(w)
            if not token:
                time.sleep(30); continue
            env["AWP_WALLET_TOKEN"] = token
            last_refresh = time.time()

            # Re-register kalau hilang
            if not w.get("registered"):
                log(name, "Re-register on-chain...")
                derive_idx = w["wallet_index"] if WALLET_MODE == "single" else None
                ok = register_wallet_onchain(name, w["address"], w["mnemonic"], derive_idx)
                if ok: w["registered"] = True
                else:
                    time.sleep(60); continue

            # Start worker
            cs = agent_status(py_bin, wd, env)
            if cs.get("state") != "running":
                r = agent_start(py_bin, wd, env)
                state = r.get("state", "?"); msg = r.get("user_message", "")
                if "not registered" in msg.lower():
                    w["registered"] = False
                    log(name, "⚠ Still not registered on-chain?")
                    time.sleep(20); continue
                if state == "selection_required":
                    use_ids = DATASET_IDS if DATASET_IDS else []
                    if not use_ids:
                        dss = r.get("datasets",[]) or r.get("_internal",{}).get("datasets",[])
                        use_ids = [str(d.get("dataset_id") or d.get("id") or "") for d in dss if d]
                    if use_ids:
                        r = agent_start(py_bin, wd, env, ds=use_ids)
                        state = r.get("state","?")
                if state == "running":
                    log(name, "Agent ✓ running")
                else:
                    log(name, f"Start: state={state} {r.get('user_message','')[:80]}")
                    time.sleep(30); continue
                time.sleep(10)

            # Monitor
            fail = 0
            while True:
                time.sleep(MONITOR_INTERVAL)
                if time.time() - last_refresh > TOKEN_REFRESH_INTERVAL:
                    nt = unlock_token(w)
                    if nt:
                        env["AWP_WALLET_TOKEN"] = nt
                        last_refresh = time.time()
                cs = agent_status(py_bin, wd, env)
                st = cs.get("state", "?"); msg = cs.get("user_message", "")
                if st == "running":
                    i = cs.get("_internal",{}); sd = i.get("status",{})
                    tot = sd.get("session_totals",{})
                    log(name, f"OK ep={sd.get('epoch_submitted',0)}/{sd.get('epoch_target',80)} "
                              f"proc={tot.get('processed_items',0)} sub={tot.get('submitted_items',0)} "
                              f"credit={sd.get('credit_score','?')}")
                    fail = 0
                elif "not registered" in msg.lower():
                    w["registered"] = False; break
                else:
                    fail += 1
                    log(name, f"State={st} ({fail}/{MAX_FAIL_BEFORE_RESTART}) {msg[:80]}")
                    if fail >= MAX_FAIL_BEFORE_RESTART:
                        agent_stop(py_bin, wd, env); time.sleep(5); break
        except Exception as e:
            log(name, f"Err: {e}, retry 30s")
            time.sleep(30)

def main():
    banner(f"Mine Bot MULTI — ON-CHAIN REGISTER | {WALLET_MODE} | {WALLET_COUNT}")
    print(f"  AWP Registry: {AWP_REGISTRY_CONTRACT}")
    print(f"  Chain: Base Mainnet ({AWP_CHAIN_ID})")
    print(f"  RPC: {BASE_RPC_URL}")

    raw_w = load_wallets()
    print(f"\n  {len(raw_w)} wallet:")
    for n, m, idx in raw_w:
        print(f"  • {n}" + (f" (hd {idx})" if idx is not None else ""))

    ensure_python311()
    install_deps()
    if USE_PROXY: test_proxy()

    # Check funder
    banner("Step 4: Check Funder Wallet")
    funder_ok = check_funder_balance()
    if not funder_ok:
        print("\n  ⚠ Continuing without funder — setRecipient akan FAIL kecuali wallet sudah punya ETH")
        answer = input("  Lanjutkan? (y/n): ").strip().lower()
        if answer != 'y':
            print("  Aborted.")
            sys.exit(0)

    prepare_template_venv()

    banner(f"Setting up {len(raw_w)} wallets (batch {BATCH_SIZE})")
    all_info = []
    args = [(n, m, (idx if idx is not None else i), len(raw_w)) 
            for i, (n, m, idx) in enumerate(raw_w)]
    
    for bs in range(0, len(args), BATCH_SIZE):
        batch = args[bs:bs+BATCH_SIZE]
        bn = bs//BATCH_SIZE + 1; tb = (len(args)+BATCH_SIZE-1)//BATCH_SIZE
        print(f"\n  Batch {bn}/{tb} — wallet {bs+1}–{bs+len(batch)}", flush=True)
        res = [None]*len(batch); th = []
        for i, a in enumerate(batch):
            t = threading.Thread(target=lambda a=a, idx=i: res.__setitem__(idx, setup_wallet_worker(a)))
            t.start(); th.append(t); time.sleep(0.3)
        for t in th: t.join()
        all_info.extend([r for r in res if r])
        print(f"  Batch {bn} done: {sum(1 for r in res if r)}/{len(batch)}")
        if bs+BATCH_SIZE < len(args): time.sleep(3)

    banner("Setup Summary")
    print(f"  {'Name':12} {'Address':44} {'IP':16} {'Reg':4} Status")
    for w in all_info:
        s = "OK" if w["token"] else "F"
        r = "✓" if w.get("registered") else "✗"
        print(f"  {w['name']:12} {w['address']:44} {w['ip']:16} {r:4} [{s}]")
    
    ok = sum(1 for w in all_info if w["token"])
    reg = sum(1 for w in all_info if w.get("registered"))
    print(f"\n  Total: {ok} unlocked | {reg} registered | dari {len(raw_w)}")
    if not all_info: sys.exit(1)

    banner(f"Starting Mining — {ok} wallets")
    threads = []
    for w in all_info:
        if not w["token"]: continue
        t = threading.Thread(target=start_mining, args=(w,), name=w["name"][:12], daemon=True)
        t.start(); threads.append(t)
        log(w["name"], f"Mining thread ✓ (IP {w['ip']})"); time.sleep(0.5)

    try:
        while True:
            time.sleep(120)
            alive = sum(1 for t in threads if t.is_alive())
            with print_lock:
                print(f"\n[{datetime.now().strftime('%H:%M:%S')}] Active: {alive}/{len(threads)}")
    except KeyboardInterrupt:
        print("\nStopping...")
        for w in all_info:
            if w["token"]:
                try:
                    e = set_env(w["agent_id"], w["name"])
                    e["AWP_WALLET_TOKEN"] = w["token"]
                    agent_stop(w["python_bin"], w["directory"], e)
                except Exception: pass
        sys.exit(0)
